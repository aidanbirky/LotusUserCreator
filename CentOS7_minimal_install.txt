
## The basics ##

yum update -y && yum upgrade -y

yum install epel-release policycoreutils-python bash-completion nano git tmux ntp iptables-services

systemctl start ntpd && systemctl enable ntpd

systemctl start iptables && systemctl enable iptables
systemctl start ip6tables && systemctl enable ip6tables

yum remove firewalld
systemctl mask firewalld

iptables -F

timedatectl set-timezone GMT


## Create swapfile ##

fallocate -l 2G /swapfile
chmod 600 /swapfile
mkswap /swapfile
swapon /swapfile
sh -c 'echo "/swapfile none swap sw 0 0" >> /etc/fstab'


## Configure new user ##

adduser user
passwd user
gpasswd -a user wheel


## Configure SSHD ##

semanage port -a -t ssh_port_t -p tcp 2233

/etc/ssh/sshd_config
Port 2233
PermitRootLogin no

systemctl restart sshd

ssh-keygen -t restart
ssh-copy-id user@ip address
